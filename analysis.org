* Overview

We extract activations from Qwen 3 32B for fictional characters. Each character is prompted with 240 questions, and we average the layer 32 activations across all questions to get a single 5120-dimensional vector per character.

PCA on this space reveals interpretable axes of variation.

This work builds on the methodology from [[https://arxiv.org/abs/2601.10387][Lu et al. (2026), "The Assistant Axis"]].

** Prompting Strategy

Role prompts are minimal: just the character name and source (e.g., "You are Harry Potter from Harry Potter."). We intentionally omit character descriptions to let the model's own representation emerge.

def analyze_universe_residuals(universe_name, prefix, n_pcs=5):
    """PCA on residuals within a single universe."""
    indices = get_universe_indices(prefix)
    if len(indices) < 10:
        print(f"{universe_name}: Only {len(indices)} characters, skipping")
        return
    
    u_residuals = residuals[indices]
    u_names = [char_names[i] for i in indices]
    
    # PCA on this universe's residuals
    u_pca = SkPCA(n_components=min(n_pcs, len(indices)-1))
    u_transformed = u_pca.fit_transform(u_residuals)
    
    print(f"=== {universe_name} ({len(indices)} characters) ===\n")
    
    rows = []
    for pc_idx in range(min(n_pcs, u_pca.n_components_)):
        scores = u_transformed[:, pc_idx]
        sorted_idx = np.argsort(scores)
        var_pct = u_pca.explained_variance_ratio_[pc_idx] * 100

            'PC': pc_idx + 1,
            'Var%': f"{var_pct:.1f}%",
            'High': ', '.join(high),
            'Low': ', '.join(low),
        })
    
    df = pd.DataFrame(rows)

def analyze_universe_full(universe_name, prefix, n_pcs=5):

    indices = get_universe_indices(prefix)
    if len(indices) < 10:

        return
    
    u_activations = activation_matrix[indices]
    u_names = [char_names[i] for i in indices]

        rows.append({
            'PC': pc_idx + 1,
            'Var%': f"{var_pct:.1f}%",
            'High': ', '.join(high),
            'Low': ', '.join(low),
        })
    
    df = pd.DataFrame(rows)
    print(df.to_string(index=False))
    print()

* Sanity Check: Similar Characters

Basic check that the embedding space is sensible - similar characters should cluster.

#+begin_src python :exports results
def find_similar(query_name, top_k=5):
    query_idx = None
    for i, name in enumerate(char_names):
        if query_name.lower() in name.lower():
            query_idx = i
            break
    
    if query_idx is None:
        return f"'{query_name}' not found"
    
    query_vec = transformed[query_idx:query_idx+1]
    distances = cdist(query_vec, transformed, metric='euclidean')[0]
    sorted_idx = np.argsort(distances)
    
    results = []
    for i in sorted_idx[1:top_k+1]:
        results.append(char_names[i].replace('_', ' ').title())
    return ', '.join(results)

if 'pca' in dir():
    checks = ['harry_potter', 'darth_vader', 'hamlet', 'zeus', 'naruto']
    for c in checks:
        print(f"{c.replace('_', ' ').title()}: {find_similar(c)}")
