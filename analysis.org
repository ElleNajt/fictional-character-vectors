* Overview

We extracted activations from Qwen 2.5 32B for 399 fictional characters. Each character was prompted with 240 questions, and we averaged the layer 32 activations across all questions to get a single 5120-dimensional vector per character.

PCA on this space reveals interpretable axes of variation.

This work builds on the methodology from [[https://arxiv.org/abs/2601.10387][Lu et al. (2026), "The Assistant Axis: Situating and Stabilizing the Default Persona of Language Models"]], which extracted persona vectors for 275 generic character archetypes.

def show_pc(pc_idx, top_k=5):
    """Show top and bottom characters for a PC."""
    scores = transformed[:, pc_idx]
    sorted_idx = np.argsort(scores)
    var_pct = pca.explained_variance_ratio_[pc_idx] * 100
    
    print(f"\n=== PC{pc_idx+1} ({var_pct:.1f}% variance) ===")
    print("\nHIGH:")
    for i in sorted_idx[-top_k:][::-1]:
        name = char_names[i].replace('_', ' ').title()
        print(f"  {name:35s} {scores[i]:+.1f}")
    print("\nLOW:")
    for i in sorted_idx[:top_k]:
        name = char_names[i].replace('_', ' ').title()
        print(f"  {name:35s} {scores[i]:+.1f}")

for i in range(10):

from scipy.spatial.distance import cdist

def find_similar(query_name, top_k=10):
    """Find characters most similar to query in PCA space."""
    # Find query index
    query_idx = None
    for i, name in enumerate(char_names):
        if query_name.lower() in name.lower():
            query_idx = i
            break
    
    if query_idx is None:
        print(f"Character '{query_name}' not found")
        return
    
    query_vec = transformed[query_idx:query_idx+1]
    distances = cdist(query_vec, transformed, metric='euclidean')[0]
    
    sorted_idx = np.argsort(distances)
    
    print(f"\nCharacters most similar to {char_names[query_idx].replace('_', ' ').title()}:\n")
    for i in sorted_idx[1:top_k+1]:  # Skip self
        name = char_names[i].replace('_', ' ').title()
        print(f"  {name:35s} (distance: {distances[i]:.1f})")

** Character Arithmetic

Like word2vec's "king - man + woman = queen", we can do character arithmetic.

#+begin_src python
def char_arithmetic(positive, negative, top_k=5):
    """
    Find characters closest to: sum(positive) - sum(negative)
    
    Example: villain + comedy = ?
    """
    result_vec = np.zeros(transformed.shape[1])
    
    used_indices = set()
    
    for name in positive:
        for i, cname in enumerate(char_names):
            if name.lower() in cname.lower():
                result_vec += transformed[i]
                used_indices.add(i)
                break
    
    for name in negative:
        for i, cname in enumerate(char_names):
            if name.lower() in cname.lower():
                result_vec -= transformed[i]
                used_indices.add(i)
                break
    
    distances = cdist(result_vec.reshape(1, -1), transformed, metric='euclidean')[0]
    sorted_idx = np.argsort(distances)
    
    pos_str = ' + '.join(positive)
    neg_str = ' - '.join(negative) if negative else ''
    equation = pos_str + (' - ' + neg_str if neg_str else '')
    
    print(f"\n{equation} = ?\n")
    count = 0
    for i in sorted_idx:
        if i not in used_indices:
            name = char_names[i].replace('_', ' ').title()
            print(f"  {name:35s} (distance: {distances[i]:.1f})")
            count += 1
            if count >= top_k:
                break

# Darth Vader - evil = ? (should give heroic Star Wars characters)
char_arithmetic(["darth_vader"], ["voldemort"])

def get_franchise_centroid(franchise_chars):
    """Get centroid of franchise characters in PCA space."""
    indices = []
    for fc in franchise_chars:
        for i, name in enumerate(char_names):
            if fc.lower() in name.lower():
                indices.append(i)
                break
    if not indices:
        return None, 0
    return transformed[indices].mean(axis=0), len(indices)

# Calculate inter-franchise distances
franchise_names = list(franchises.keys())
centroids = {}
for fname, chars in franchises.items():
    centroid, count = get_franchise_centroid(chars)
    if centroid is not None:
        centroids[fname] = centroid
        print(f"{fname}: {count} characters found")

print("\n=== Inter-Franchise Distances ===\n")
for i, f1 in enumerate(franchise_names):
    if f1 not in centroids:
        continue
    for f2 in franchise_names[i+1:]:
        if f2 not in centroids:
            continue
        dist = np.linalg.norm(centroids[f1] - centroids[f2])
        print(f"  {f1:20s} <-> {f2:20s}: {dist:.1f}")

* Comparison to Assistant Axis Role Vectors

Lu et al. extracted vectors for 463 generic roles using the same methodology. When we project fictional characters onto their role-based PCA:

#+begin_src python
# This requires the Christina/Assistant Axis vectors
# Load if available
christina_path = Path('../MindViruses/christina_vectors_backup/layer32_pos23.pt')
if christina_path.exists():
    import torch
    christina = torch.load(christina_path, map_location='cpu', weights_only=False)
    christina_pca = christina['pca']
    christina_scaler = christina['scaler']
    
    # Project our characters onto Christina's space
    scaled = christina_scaler.transform(activation_matrix)
    projected = christina_pca.transform(scaled)
    
    # Calculate variance explained
    total_var = np.var(scaled, axis=0).sum()
    var_per_comp = np.var(projected, axis=0)
    
    print("Character variance explained by Assistant Axis role PCs:\n")
    cumvar = 0
    for i in [1, 5, 10, 20, 50, 100]:
        if i <= len(var_per_comp):
            cumvar = sum(var_per_comp[:i]) / total_var
            print(f"  {i:3d} components: {cumvar:.1%}")
else:
    print("Assistant Axis vectors not found - skipping comparison")
#+end_src

*Finding*: The fictional character space is largely orthogonal to the generic role space. Role-based PCs only capture ~43% of fictional character variance with 100 components, while character-specific PCA captures 56% with just 10.

Characters are more than the sum of their role labels—they carry additional structure from their specific narratives, relationships, and cultural contexts.
