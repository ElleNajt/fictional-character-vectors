* Overview

We extract activations from Qwen 3 32B for fictional characters. Each character is prompted with 240 questions, and we average the layer 32 activations across all questions to get a single 5120-dimensional vector per character.

PCA on this space reveals interpretable axes of variation.

This work builds on the methodology from [[https://arxiv.org/abs/2601.10387][Lu et al. (2026), "The Assistant Axis"]].

** Prompting Strategy

Role prompts are minimal: just the character name and source (e.g., "You are Harry Potter from Harry Potter."). We intentionally omit character descriptions to let the model's own representation emerge.

def show_pc(pc_idx, top_k=5):
    """Show top and bottom characters for a PC."""
    scores = transformed[:, pc_idx]
    sorted_idx = np.argsort(scores)
    var_pct = pca.explained_variance_ratio_[pc_idx] * 100
    
    high = [char_names[i].replace('_', ' ').title() for i in sorted_idx[-top_k:][::-1]]
    low = [char_names[i].replace('_', ' ').title() for i in sorted_idx[:top_k]]
    
    return {
        'PC': pc_idx + 1,
        'Var%': f"{var_pct:.1f}%",
        'High': ', '.join(high[:3]),
        'Low': ', '.join(low[:3])
    }

if 'pca' in dir():

def find_similar(query_name, top_k=10):
    query_idx = None
    for i, name in enumerate(char_names):
        if query_name.lower() in name.lower():
            query_idx = i
            break
    
    if query_idx is None:
        return f"Character '{query_name}' not found"
    
    query_vec = transformed[query_idx:query_idx+1]
    distances = cdist(query_vec, transformed, metric='euclidean')[0]
    sorted_idx = np.argsort(distances)
    
    rows = []
    for i in sorted_idx[1:top_k+1]:
        name = char_names[i].replace('_', ' ').title()
        rows.append({'Character': name, 'Distance': f"{distances[i]:.1f}"})
    
    return pd.DataFrame(rows)

* Character Arithmetic

Like word2vec's "king - man + woman = queen".

#+begin_src python :exports results
def char_arithmetic(positive, negative, top_k=5):
    result_vec = np.zeros(transformed.shape[1])
    used_indices = set()
    
    for name in positive:
        for i, cname in enumerate(char_names):
            if name.lower() in cname.lower():
                result_vec += transformed[i]
                used_indices.add(i)
                break
    
    for name in negative:
        for i, cname in enumerate(char_names):
            if name.lower() in cname.lower():
                result_vec -= transformed[i]
                used_indices.add(i)
                break
    
    distances = cdist(result_vec.reshape(1, -1), transformed, metric='euclidean')[0]
    sorted_idx = np.argsort(distances)
    
    rows = []
    count = 0
    for i in sorted_idx:
        if i not in used_indices:
            name = char_names[i].replace('_', ' ').title()
            rows.append({'Character': name, 'Distance': f"{distances[i]:.1f}"})
            count += 1
            if count >= top_k:
                break
    
    return pd.DataFrame(rows)

if 'pca' in dir():
    print("Darth Vader - Voldemort = ? (heroic Star Wars)")
    print(char_arithmetic(["darth_vader"], ["voldemort"]).to_string(index=False))

* Comparison to Role Vectors

Lu et al. extracted vectors for 463 generic roles (e.g., "accountant", "hero", "villain"). We project characters onto this space and analyze residuals.

#+begin_src python :exports results
from sklearn.preprocessing import StandardScaler

# Check for role data
role_pca_path = Path('data/role_vectors/qwen-3-32b_pca_layer32.pkl')

if role_pca_path.exists() and 'activation_matrix' in dir():
    with open(role_pca_path, 'rb') as f:
        role_data = pickle.load(f)
    
    role_pca = role_data['pca']
    role_scaler = role_data['scaler']
    roles_flat = role_data['role_names']
    role_projections = role_data['transformed']
    
    # Project characters onto role space
    chars_scaled = role_scaler.transform(activation_matrix)
    chars_in_role_space = role_pca.transform(chars_scaled)
    
    # Reconstruct and get residuals
    reconstructed = chars_in_role_space @ role_pca.components_
    residuals = chars_scaled - reconstructed
    
    original_var = np.var(chars_scaled, axis=0).sum()
    residual_var = np.var(residuals, axis=0).sum()
    captured_var = original_var - residual_var
    
    df = pd.DataFrame({
        'Metric': ['Roles loaded', 'Role PCs', 'Captured by roles', 'Residual (not captured)'],
        'Value': [len(roles_flat), role_pca.n_components_, 
                  f"{captured_var/original_var:.1%}", f"{residual_var/original_var:.1%}"]
    })
    print(df.to_string(index=False))
else:
    print("Role vectors not found. Download from HuggingFace (see README).")

def show_role_pc(pc_idx, top_k=3):
    role_scores = role_projections[:, pc_idx]
